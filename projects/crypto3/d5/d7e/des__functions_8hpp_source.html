37  inline static void ip(word_type &L, word_type &R, const block_type &block) {
38  L = boost::endian::native_to_big(block[0]);
39  R = boost::endian::native_to_big(block[1]);
40 
41  word_type T;
42  R = policy_type::template rotl<4>(R);
43  T = (L ^ R) & 0xF0F0F0F0;
44  L ^= T;
45  R = policy_type::template rotr<20>(R ^ T);
46  T = (L ^ R) & 0xFFFF0000;
47  L ^= T;
48  R = policy_type::template rotr<18>(R ^ T);
49  T = (L ^ R) & 0x33333333;

62  word_type T;
63

144  inline static void des_encrypt(word_type &L, word_type &R, const key_schedule_type &round_key) {
145  for (size_t i = 0; i != policy_type::rounds; i += 2) {
146  L ^= spbox(policy_type::template rotr<4>(R) ^ round_key[2 * i], R ^ round_key[2 * i + 1]);
147  R ^= spbox(policy_type::template rotr<4>(L) ^ round_key[2 * i + 2],
148  L ^ round_key[2 * i + 3]);
149  }

152  inline static void des_decrypt(word_type &L, word_type &R, const key_schedule_type &round_key) {
153  for (size_t i = policy_type::rounds; i != 0; i -= 2) {
154  L ^= spbox(policy_type::template rotr<4>(R) ^ round_key[2 * i - 2],
155  R ^ round_key[2 * i - 1]);
