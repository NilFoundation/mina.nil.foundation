81  void process_block() {

83  block_type block;
84  ::nil::crypto3::detail::pack<endian_type, value_bits, word_bits>(value_array, block);

122  std::size_t i = seen % block_bits;

187  digest_type end_message() {
188  using namespace nil::crypto3::detail;
189 
190  length_type length = seen;
191 
192  // Add a 1 bit
193 #ifdef CRYPTO3_HASH_NO_OPTIMIZATION
194  std::array<bool, ValueBits> padding_bits = {{1}};
195  std::array<value_type, 1> padding_values;
196  ::nil::crypto3::detail::pack<endian, 1, ValueBits>(padding_bits, padding_values);
197  update_one(padding_values[0]);
198 #else
199  value_type pad = 0;
200  ::nil::crypto3::detail::imploder_step<endian_type, 1, value_bits, 0>::step(1, pad);
201  update_one(pad);
202 #endif
203 
204  // Pad with 0 bits
205  while ((seen + length_bits) % block_bits != 0) {
206  update_one(value_type());
207  }
208 
209  // Append length
210  append_length<int>(length);
211 
212  // Reset for next message
213  seen = 0;
214 
215  // Calculate static_digest and reset block_hash
216  return block_hash.end_message();
217  }
