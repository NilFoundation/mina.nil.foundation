55  void begin_message(const block_type &block) {

58  if (nonce_len == 12) {

72  void process_block(const block_type &block) {
73  if (m_aad_buf_pos > 0) {
74  const size_t taking = std::min(GCM_BS - m_aad_buf_pos, size);
75  copy_mem(&m_aad_buf[m_aad_buf_pos], input, taking);
76  m_aad_buf_pos += taking;
77  input += taking;
78  size -= taking;
79 
80  if (m_aad_buf_pos == GCM_BS) {
81  hash->update_associated_data(m_aad_buf.data(), GCM_BS);
82  m_aad_buf_pos = 0;
83  }
84  }
85 
86  const size_t left_over = size % GCM_BS;
87  const size_t full_blocks = size - left_over;
88  hash->update_associated_data(input, full_blocks);
89  input += full_blocks;
90 
91  if (left_over > 0) {
92  copy_mem(&m_aad_buf[m_aad_buf_pos], input, left_over);
93  m_aad_buf_pos += left_over;
94  }
95  }
