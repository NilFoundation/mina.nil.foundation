61  static inline void shift_rows(block_type &state, const shift_offsets_type &offset) {
62  std::array<typename policy_type::byte_type, policy_type::block_words> tmp = {0};
63 
64  // row 0 never gets shifted
65 #pragma clang loop unroll(full)
66  for (std::size_t row = 1; row < policy_type::word_bytes; ++row) {
67  const std::size_t off = offset[row - 1];
68 #pragma clang loop unroll(full)
69  for (std::size_t i = 0; i < off; ++i) {
70  tmp[i] = state[i * policy_type::word_bytes + row];
71  }
72 #pragma clang loop unroll(full)
73  for (std::size_t i = 0; i < policy_type::block_words - 1; ++i) {
74  state[i * policy_type::word_bytes + row] =
75  state[(i + off) * policy_type::word_bytes + row];
76  }
77 #pragma clang loop unroll(full)
78  for (std::size_t i = 0; i < off; ++i) {
79  state[(policy_type::block_words - 1 - i) * policy_type::word_bytes + row] =
80  tmp[off - 1 - i];
81  }
82  }

87  block_type tmp = {0};
88 
89 #pragma clang loop unroll(full)
90  for (std::size_t col = 0; col < policy_type::block_words; ++col) {
91 #pragma clang loop unroll(full)
92  for (std::size_t row = 0; row < policy_type::word_bytes; ++row) {
93 #pragma clang loop unroll(full)

95  tmp[col * policy_type::word_bytes + row] ^=

105  template<typename InputIterator>
106  static inline void add_round_key(block_type &state, InputIterator first, InputIterator last) {
107  BOOST_ASSERT(std::distance(first, last) == policy_type::block_words &&
108  state.size() == policy_type::block_bytes);
109 
110 #pragma clang loop unroll(full)
111  for (std::size_t i = 0; i < policy_type::block_words && first != last; ++i && ++first) {
112 #pragma clang loop unroll(full)
113  for (std::size_t j = 0; j < policy_type::word_bytes; ++j) {
114  state[i * policy_type::word_bytes + j] ^=
115  ::nil::crypto3::detail::extract_uint_t<CHAR_BIT>(*first,
116  policy_type::word_bytes - (j + 1));
117  }
118  }

172  static void schedule_key(const key_type &key, key_schedule_type &encryption_key,
173  key_schedule_type &decryption_key) {
174  // the first key_words words are the original key
175  ::nil::crypto3::detail::pack<stream_endian::little_octet_big_bit, CHAR_BIT,
176  policy_type::word_bits>(
177  key.begin(), key.begin() + policy_type::key_words * policy_type::word_bytes,
178  encryption_key.begin(), encryption_key.begin() + policy_type::key_words);
179 
180 #pragma clang loop unroll(full)
181  for (std::size_t i = policy_type::key_words; i < policy_type::key_schedule_words; ++i) {
182  typename policy_type::key_schedule_word_type tmp = encryption_key[i - 1];
183  if (i % policy_type::key_words == 0) {
184  tmp = sub_word(policy_type::rotate_left(tmp), policy_type::constants) ^
185  policy_type::round_constants[i / policy_type::key_words - 1];
186  } else if (policy_type::key_words > 6 && i % policy_type::key_words == 4) {
187  tmp = sub_word(tmp, policy_type::constants);
188  }
189  encryption_key[i] = encryption_key[i - policy_type::key_words] ^ tmp;
190  }
191 
192  std::array<typename policy_type::byte_type, policy_type::key_schedule_bytes> bekey = {0};
193  ::nil::crypto3::detail::pack<stream_endian::little_octet_big_bit, policy_type::word_bits,
194  CHAR_BIT>(encryption_key, bekey);
195 
196 #pragma clang loop unroll(full)
197  for (std::uint8_t round = 1; round < policy_type::rounds; ++round) {
198  move(mix_columns(boost::adaptors::slice(bekey, round * policy_type::block_bytes,
199  (round + 1) * policy_type::block_bytes),
200  policy_type::inverted_mm),
201  bekey.begin() + round * policy_type::block_bytes);
202  }
203 
204  ::nil::crypto3::detail::pack<stream_endian::little_octet_big_bit, CHAR_BIT,
205  policy_type::word_bits>(bekey, decryption_key);
206  }
