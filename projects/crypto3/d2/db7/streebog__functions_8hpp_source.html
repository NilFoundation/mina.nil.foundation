38  inline static void addm(const uint8_t *m, word_type *h) {
39  word_type carry = 0;
40  for (int i = 0; i < block_words; i++) {
41  const word_type m64 = boost::endian::native_to_little(m[i]);
42  const word_type hi = boost::endian::native_to_little(reinterpret_cast<uint8_t *>(h)[i]);
43  const word_type t = hi + m64;
44 
45  const word_type overflow = (t < hi ? 1 : 0) | (t < m64 ? 1 : 0);
46  store_le(t + carry, reinterpret_cast<uint8_t *>(&h[i]));
47  carry = overflow;
48  }
49  }
