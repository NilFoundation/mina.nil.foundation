92  inline block_type encrypt_block(const block_type &plaintext) const {
93  word_type d1 = boost::endian::native_to_big(plaintext[0]);
94  word_type d2 = boost::endian::native_to_big(plaintext[1]);
95 
96  const uint64_t *K = key_schedule.data();
97 
98  d1 ^= *K++;
99  d2 ^= *K++;
100 
101  d2 ^= policy_type::f_slow(d1, *K++);
102  d1 ^= policy_type::f_slow(d2, *K++);
103 
104  for (size_t r = 1; r != rounds / 2 - 1; ++r) {
105  if (r % 3 == 0) {
106  d1 = policy_type::fl(d1, *K++);
107  d2 = policy_type::flinv(d2, *K++);
108  }
109 
110  d2 ^= policy_type::f(d1, *K++);
111  d1 ^= policy_type::f(d2, *K++);
112  }
113 
114  d2 ^= policy_type::f_slow(d1, *K++);
115  d1 ^= policy_type::f_slow(d2, *K++);
116 
117  d2 ^= *K++;
118  d1 ^= *K++;
119 
120  return {boost::endian::big_to_native(d2), boost::endian::big_to_native(d1)};
121  }
