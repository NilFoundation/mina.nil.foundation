template<typename Hash, typename StateAccumulator, typename Params>
class nil::crypto3::hash::haifa_stream_processor< Hash, StateAccumulator, Params >

This will do the usual HAIFA-style strengthening, padding with a 1 bit, then 0 bits as needed, then, if requested, the length and the digest bits.
