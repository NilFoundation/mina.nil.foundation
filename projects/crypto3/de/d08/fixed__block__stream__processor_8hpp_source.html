64  void update_one(value_type value) {
65  std::size_t i = seen % input_block_bits;
66  cache[i / value_bits] = value;
67  seen += value_bits;
68  if (i == input_block_bits - value_bits) {
69  // Convert the input into words
70  input_block_type block = {0};
71  ::nil::crypto3::detail::pack<endian_type, value_bits, input_value_bits>(cache.begin(),
72  cache.end(), block);
73 
74  // Process the block
75  std::size_t ibb = input_block_bits;
76  state(block, accumulators::bits = ibb);
77 
78  // Reset seen if we don't need to track the length
79  if (!length_bits) {
80  seen = 0;
81  }
82  }
83  }

85  template<typename InputIterator>
86  void update_n(InputIterator first, InputIterator last) {
87  std::size_t n = std::distance(first, last);
88 #ifndef CRYPTO3_CODEC_NO_OPTIMIZATION
89 #pragma clang loop unroll(full)
90  for (; n && (seen % input_block_bits); --n, ++first) {
91  update_one(*first);
92  }
93 #pragma clang loop unroll(full)
94  for (; n >= block_values; n -= block_values, first += block_values) {
95  // Convert the input into words
96  input_block_type block = {0};
97  ::nil::crypto3::detail::pack<endian_type, value_bits, input_value_bits>(
98  first, first + block_values, block);
99  seen += value_bits * block_values;
100 
101  std::size_t ibb = input_block_bits;
102  state(block, accumulators::bits = ibb);
103 
104  // Reset seen if we don't need to track the length
105  if (!length_bits) {
106  seen = 0;
107  }
108  }
109 #endif
110 
111 #pragma clang loop unroll(full)
112  for (; n; --n, ++first) {
113  update_one(*first);
114  }
115  }
