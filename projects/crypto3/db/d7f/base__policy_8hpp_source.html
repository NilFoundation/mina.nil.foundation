191  out[4] = constants()[((plaintext[2] & 0x0FU) << 1) | (plaintext[3] >> 7)];
192  out[5] = constants()[((plaintext[3] & 0x7CU) >> 2)];
193  out[6] = constants()[((plaintext[3] & 0x03U) << 3) | (plaintext[4] >> 5)];

199  static inline decoded_block_type decode_block(const encoded_block_type &encoded) {
200  decoded_block_type out = {0};
201  encoded_block_type output_buffer = {0};
202 
203  auto oit = std::begin(output_buffer);
204 
205  for (auto it = std::begin(encoded); it != std::end(encoded); ++it) {
206  const uint8_t bin = inverted_constants()[*it];
207 
208  if (bin <= 0x3f) {
209  *oit++ = bin;
210  } else if (!(bin == 0x81 || bin == 0x80)) {
211  BOOST_THROW_EXCEPTION(non_base_input<32>());
212  }
213 
214  /*
215  * If we're at the end of the input, pad with 0s and truncate
216  */
217  if (std::distance(it, encoded.end()) == 1 && std::distance(output_buffer.begin(), oit)) {
218  for (auto itr = oit;
219  std::distance(output_buffer.begin(), itr) < decoded_block_bits / CHAR_BIT;
220  ++itr) {
221  *itr = 0x00U;
222  }
223 
224  oit = output_buffer.end();
225  }
226 
227  if (oit == output_buffer.end()) {
228  out[0] = (output_buffer[0] << 3U) | (output_buffer[1] >> 2U);
229  out[1] = (output_buffer[1] << 6U) | (output_buffer[2] << 1U) | (output_buffer[3] >> 4U);
230  out[2] = (output_buffer[3] << 4U) | (output_buffer[4] >> 1U);
231  out[3] = (output_buffer[4] << 7U) | (output_buffer[5] << 2U) | (output_buffer[6] >> 3U);
232  out[4] = (output_buffer[6] << 5U) | output_buffer[7];
233 
234  oit = output_buffer.begin();
235  }
236  }
237 
238  return out;
239  }

297  template<typename NumberType = cpp_int>
298  static inline encoded_block_type encode_block(const decoded_block_type &plaintext) {
299  encoded_block_type out;
300  std::size_t csint = basic_base_policy<58>::constants_size;
301  NumberType v = 0, q = 0, r = 0, cs(csint);
302 
303  import_bits(v, plaintext.begin(), plaintext.end(), encoded_value_bits);
304  while (v != 0) {
305  divide_qr(v, cs, q, r);
306  out.emplace_back(constants()[r.template convert_to<std::uint8_t>()]);
307  v = q;
308  }
309  return out;
310  }

312  template<typename NumberType = cpp_int>
313  static inline decoded_block_type decode_block(const encoded_block_type &plaintext) {
314  decoded_block_type out;
315  std::size_t csint = basic_base_policy<58>::constants_size;
316  NumberType v = 0;
317 
318  for (const typename encoded_block_type::value_type &c : plaintext) {
319  if (c == ' ' || c == '\n') {
320  continue;
321  }
322  const size_t idx = inverted_constants()[c];
323 
324  if (idx == 0x80) {
325  throw wrong_input_symbol<58>();
326  // throw std::invalid_argument("Invalid base58");
327  }
328  v *= csint;
329  v += idx;
330  }
331  export_bits(v, std::inserter(out, out.end()), decoded_value_bits);
332  std::reverse(out.begin(), out.end());
333  return out;
334  }
