76  inline static block_type format_b0(const associated_data_type &ad, const nonce_type &nonce,
77  size_t sz) {
78  block_type b0 = {static_cast<uint8_t>((ad.size() ? 64 : 0) +
79  ((tag_bits / 2 * CHAR_BIT - 1) << 3U) +
80  (length_bits / CHAR_BIT - 1))};
81 
82  copy_mem(&b0[1], nonce.data(), nonce.size());

127  block_type result = {0};

129  BOOST_ASSERT_MSG(buffer.size() >= offset, "Offset is sane");
130 
131  buffer.insert(buffer.begin() + offset, msg_buf().begin(), msg_buf().end());
132 
133  const size_t sz = buffer.size() - offset;
134  uint8_t *buf = buffer.data() + offset;
135 
136  const secure_vector<uint8_t> &ad = ad_buf();
137  BOOST_ASSERT_MSG(ad.size() % CCM_BS == 0, "AD is block size multiple");
138 
139  secure_vector<uint8_t> T(CCM_BS);
140  cipher.encrypt(policy_type::format_b0(sz), T);
141 
142  for (size_t i = 0; i != ad.size(); i += CCM_BS) {
143  xor_buf(T.data(), &ad[i], CCM_BS);
144  cipher.encrypt(T);
145  }
146 
147  secure_vector<uint8_t> C = policy_type::format_c0();
148  secure_vector<uint8_t> S0(CCM_BS);
149  cipher.encrypt(C, S0);
150  policy_type::inc(C);
151 
152  secure_vector<uint8_t> X(CCM_BS);
153 
154  const uint8_t *buf_end = &buf[sz];
155 
156  while (buf != buf_end) {
157  const size_t to_proc = std::min<size_t>(CCM_BS, buf_end - buf);
158 
159  xor_buf(T.data(), buf, to_proc);
160  cipher.encrypt(T);
161 
162  cipher.encrypt(C, X);
163  xor_buf(buf, X.data(), to_proc);
164  policy_type::inc(C);
165 
166  buf += to_proc;
167  }
168 
169  T ^= S0;
170 
171  buffer += std::make_pair(T.data(), tag_bits / CHAR_BIT);
172 
173  return result;
174  }
