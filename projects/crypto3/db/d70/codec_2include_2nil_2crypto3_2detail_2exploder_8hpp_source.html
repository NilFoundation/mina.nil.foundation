58  int const kb = (k % UnitBits);

115  // To keep the implementation managable, input and output sizes must
116  // be multiples or factors of the unit size.
117  // If one of these is firing, you may want a bit-only stream_endian
118  // rather than one that mentions bytes or octets.
119  BOOST_STATIC_ASSERT(!(InputBits % UnitBits && UnitBits % InputBits));
120  BOOST_STATIC_ASSERT(!(OutputBits % UnitBits && UnitBits % OutputBits));
