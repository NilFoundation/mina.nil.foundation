82  inline static block_type unpad(const block_type &block, size_type bytes_in_block) {
83  ct::poison(block, bytes_in_block);
84  uint8_t bad_input = 0;
85  uint8_t seen_one = 0;
86  size_t pad_pos = bytes_in_block - 1;
87  size_t i = bytes_in_block;
88 
89  while (i) {
90  seen_one |= ct::is_equal<uint8_t>(block[i - 1], 0x80);
91  pad_pos -= ct::select<uint8_t>(~seen_one, 1, 0);
92  bad_input |= ~ct::is_zero<uint8_t>(block[i - 1]) & ~seen_one;
93  i--;
94  }
95  bad_input |= ~seen_one;
96 
97  ct::conditional_copy_mem(size_t(bad_input), &pad_pos, &bytes_in_block, &pad_pos, 1);
98  ct::unpoison(block, bytes_in_block);
99  ct::unpoison(pad_pos);
100  }
