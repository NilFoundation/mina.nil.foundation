119  template<typename SubstitutionType>
120  void generate_sbox(SubstitutionType &box, word_type &L, word_type &R, const salt_type &salt,
121  std::size_t salt_off) const {
122  for (size_t i = 0; i != box.size(); i += 2) {
123  if (!salt.empty()) {
124  L ^= salt[(i + salt_off) % (salt.size())];
125  R ^= salt[(i + salt_off + 1) % (salt.size())];
126  }
127 
128  for (size_t r = 0; r != policy_type::rounds; r += 2) {
129  L ^= permutations[r];
130  R ^= policy_type::bff(L, constants);
131 
132  R ^= permutations[r + 1];

134  }
