107  void schedule_key(const key_type &key) {
108  const size_t CK0 = (policy_type::key_bits / 64) - 2;
109  const size_t CK1 = (CK0 + 1) % 3;
110  const size_t CK2 = (CK1 + 1) % 3;
111 
112  word_type w0[4];
113  word_type w1[4];
114  word_type w2[4];
115  word_type w3[4];
116 
117  w0[0] = boost::endian::native_to_big(key[0]);
118  w0[1] = boost::endian::native_to_big(key[1]);
119  w0[2] = boost::endian::native_to_big(key[2]);
120  w0[3] = boost::endian::native_to_big(key[3]);
121 
122  w1[0] = w0[0] ^ policy_type::round_constants[CK0][0];
123  w1[1] = w0[1] ^ policy_type::round_constants[CK0][1];
124  w1[2] = w0[2] ^ policy_type::round_constants[CK0][2];
125  w1[3] = w0[3] ^ policy_type::round_constants[CK0][3];
126 
127  policy_type::fo(w1[0], w1[1], w1[2], w1[3]);
128 
129  if (policy_type::key_bits / 8 == 24 || policy_type::key_bits / 8 == 32) {
130  w1[0] ^= boost::endian::native_to_big(key[4]);
131  w1[1] ^= boost::endian::native_to_big(key[5]);
132  }
133  if (policy_type::key_bits / 8 == 32) {
134  w1[2] ^= boost::endian::native_to_big(key[6]);
135  w1[3] ^= boost::endian::native_to_big(key[7]);
136  }
137 
138  w2[0] = w1[0] ^ policy_type::round_constants[CK1][0];
139  w2[1] = w1[1] ^ policy_type::round_constants[CK1][1];
140  w2[2] = w1[2] ^ policy_type::round_constants[CK1][2];
141  w2[3] = w1[3] ^ policy_type::round_constants[CK1][3];
142 
143  policy_type::fe(w2[0], w2[1], w2[2], w2[3]);
144 
145  w2[0] ^= w0[0];
146  w2[1] ^= w0[1];
147  w2[2] ^= w0[2];
148  w2[3] ^= w0[3];
149 
150  w3[0] = w2[0] ^ policy_type::round_constants[CK2][0];
151  w3[1] = w2[1] ^ policy_type::round_constants[CK2][1];
152  w3[2] = w2[2] ^ policy_type::round_constants[CK2][2];
153  w3[3] = w2[3] ^ policy_type::round_constants[CK2][3];
154 
155  policy_type::fo(w3[0], w3[1], w3[2], w3[3]);
156 
157  w3[0] ^= w1[0];
158  w3[1] ^= w1[1];
159  w3[2] ^= w1[2];
160  w3[3] ^= w1[3];
161 
162  if (policy_type::key_bits / 8 == 16) {
163  encryption_round_key.resize(4 * 13);
164  } else if (policy_type::key_bits / 8 == 24) {
165  encryption_round_key.resize(4 * 15);
166  } else if (policy_type::key_bits / 8 == 32) {
167  encryption_round_key.resize(4 * 17);
168  }
169 
170  policy_type::rol128<19>(w0, w1, &encryption_round_key[0]);
171  policy_type::rol128<19>(w1, w2, &encryption_round_key[4]);
172  policy_type::rol128<19>(w2, w3, &encryption_round_key[8]);
173  policy_type::rol128<19>(w3, w0, &encryption_round_key[12]);
174  policy_type::rol128<31>(w0, w1, &encryption_round_key[16]);
175  policy_type::rol128<31>(w1, w2, &encryption_round_key[20]);
176  policy_type::rol128<31>(w2, w3, &encryption_round_key[24]);
177  policy_type::rol128<31>(w3, w0, &encryption_round_key[28]);
178  policy_type::rol128<67>(w0, w1, &encryption_round_key[32]);
179  policy_type::rol128<67>(w1, w2, &encryption_round_key[36]);
180  policy_type::rol128<67>(w2, w3, &encryption_round_key[40]);
181  policy_type::rol128<67>(w3, w0, &encryption_round_key[44]);
182  policy_type::rol128<97>(w0, w1, &encryption_round_key[48]);
183 
184  if (policy_type::key_bits / CHAR_BIT == 24 || policy_type::key_bits / CHAR_BIT == 32) {
185  policy_type::rol128<97>(w1, w2, &encryption_round_key[52]);
186  policy_type::rol128<97>(w2, w3, &encryption_round_key[56]);
187 
188  if (policy_type::key_bits / 8 == 32) {
189  policy_type::rol128<97>(w3, w0, &encryption_round_key[60]);
190  policy_type::rol128<109>(w0, w1, &encryption_round_key[64]);
191  }
192  }
193 
194  // Now the decryption key gets scheduled
195 
196  for (size_t i = 0; i != decryption_round_key.size(); i += 4) {
197  decryption_round_key[i] = encryption_round_key[encryption_round_key.size() - 4 - i];
198  decryption_round_key[i + 1] = encryption_round_key[encryption_round_key.size() - 3 - i];
199  decryption_round_key[i + 2] = encryption_round_key[encryption_round_key.size() - 2 - i];
200  decryption_round_key[i + 3] = encryption_round_key[encryption_round_key.size() - 1 - i];
201  }
202 
203  for (size_t i = 4; i != decryption_round_key.size() - 4; i += 4) {
204  for (size_t j = 0; j != 4; ++j) {
205  decryption_round_key[i + j] = policy_type::rotr<8>(decryption_round_key[i + j]) ^
206  policy_type::rotr<16>(decryption_round_key[i + j]) ^
207  policy_type::rotr<24>(decryption_round_key[i + j]);
208  }
209 
210  decryption_round_key[i + 1] ^= decryption_round_key[i + 2];
211  decryption_round_key[i + 2] ^= decryption_round_key[i + 3];
212  decryption_round_key[i + 0] ^= decryption_round_key[i + 1];
213  decryption_round_key[i + 3] ^= decryption_round_key[i + 1];
214  decryption_round_key[i + 2] ^= decryption_round_key[i + 0];
215  decryption_round_key[i + 1] ^= decryption_round_key[i + 2];
216 
217  decryption_round_key[i + 1] = ((decryption_round_key[i + 1] << 8) & 0xFF00FF00) |
218  ((decryption_round_key[i + 1] >> 8) & 0x00FF00FF);
219  decryption_round_key[i + 2] = policy_type::rotr<16>(decryption_round_key[i + 2]);
220  decryption_round_key[i + 3] = reverse_bytes(decryption_round_key[i + 3]);
221 
222  decryption_round_key[i + 1] ^= decryption_round_key[i + 2];
223  decryption_round_key[i + 2] ^= decryption_round_key[i + 3];
224  decryption_round_key[i + 0] ^= decryption_round_key[i + 1];
225  decryption_round_key[i + 3] ^= decryption_round_key[i + 1];
226  decryption_round_key[i + 2] ^= decryption_round_key[i + 0];
227  decryption_round_key[i + 1] ^= decryption_round_key[i + 2];
228  }

231  block_type transform(const block_type &plaintext, const key_schedule_type &schedule) {
232  // Hit every state line of S1 and S2
233  const size_t cache_line_size = cpuid::cache_line_size();
234 
235  /*
236  * This initializer ensures Z == 0xFFFFFFFF for any state line size
237  * in {32,64,128,256,512}
238  */
239  volatile uint32_t Z = 0x11101010;
240  for (size_t i = 0; i < policy_type::constants_size; i += cache_line_size / sizeof(uint32_t)) {
241  Z |= policy_type::s1[i] | policy_type::s2[i];
242  }
243 
244  word_type t0 = boost::endian::native_to_big(plaintext[0]);
245  word_type t1 = boost::endian::native_to_big(plaintext[1]);
246  word_type t2 = boost::endian::native_to_big(plaintext[2]);
247  word_type t3 = boost::endian::native_to_big(plaintext[3]);
248 
249  t0 &= Z;
250 
251  for (size_t r = 0; r < rounds; r += 2) {
252  t0 ^= schedule[4 * r];
253  t1 ^= schedule[4 * r + 1];
254  t2 ^= schedule[4 * r + 2];
255  t3 ^= schedule[4 * r + 3];
256  policy_type::fo(t0, t1, t2, t3);
257 
258  t0 ^= schedule[4 * r + 4];
259  t1 ^= schedule[4 * r + 5];
260  t2 ^= schedule[4 * r + 6];
261  t3 ^= schedule[4 * r + 7];
262 
263  if (r != rounds - 2) {
264  policy_type::fe(t0, t1, t2, t3);
265  }
266  }
267 
268  return {policy_type::x1[policy_type::extract_uint_t<CHAR_BIT>(t0, 0)] ^
269  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds], 0),
270  policy_type::x2[policy_type::extract_uint_t<CHAR_BIT>(t0, 1)] >> 8 ^
271  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds], 1),
272  policy_type::s1[policy_type::extract_uint_t<CHAR_BIT>(t0, 2)] ^
273  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds], 2),
274  policy_type::s2[policy_type::extract_uint_t<CHAR_BIT>(t0, 3)] ^
275  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds], 3),
276  policy_type::x1[policy_type::extract_uint_t<CHAR_BIT>(t1, 0)] ^
277  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds + 1], 0),
278  policy_type::x2[policy_type::extract_uint_t<CHAR_BIT>(t1, 1)] >> 8 ^
279  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds + 1], 1),
280  policy_type::s1[policy_type::extract_uint_t<CHAR_BIT>(t1, 2)] ^
281  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds + 1], 2),
282  policy_type::s2[policy_type::extract_uint_t<CHAR_BIT>(t1, 3)] ^
283  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds + 1], 3),
284  policy_type::x1[policy_type::extract_uint_t<CHAR_BIT>(t2, 0)] ^
285  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds + 2], 0),
286  policy_type::x2[policy_type::extract_uint_t<CHAR_BIT>(t2, 1)] >> 8 ^
287  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds + 2], 1),
288  policy_type::s1[policy_type::extract_uint_t<CHAR_BIT>(t2, 2)] ^
289  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds + 2], 2),
290  policy_type::s2[policy_type::extract_uint_t<CHAR_BIT>(t2, 3)] ^
291  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds + 2], 3),
292  policy_type::x1[policy_type::extract_uint_t<CHAR_BIT>(t3, 0)] ^
293  policy_type::extract_uint_t<CHAR_BIT>(schedule[4 * rounds + 3], 0),
