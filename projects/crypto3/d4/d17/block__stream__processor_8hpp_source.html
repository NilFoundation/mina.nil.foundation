62  void update_one(value_type value) {
63  std::size_t i = seen % input_block_bits;
64  cache[i / value_bits] = value;
65  seen += value_bits;
66  if (i == input_block_bits - value_bits) {
67  // Convert the input into words
68  input_block_type block = {0};
69  ::nil::crypto3::detail::pack<endian_type, value_bits, input_block_bits / block_values>(
70  cache.begin(), cache.end(), block);
71 
72  // Process the block
73  state(block);
74 
75  // Reset seen if we don't need to track the length
76  if (!length_bits) {
77  seen = 0;
78  }
79  }
80  }
