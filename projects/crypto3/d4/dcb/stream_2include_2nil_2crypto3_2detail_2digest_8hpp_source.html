169  template<unsigned DB>
170  std::istream &operator>>(std::istream &source, digest<DB> &d) {
171  std::array<char, DB / 4> a = {{}};
172  for (unsigned i = 0; i < a.size(); ++i) {
173  char c;
174  if (!source.get(c)) {
175  source.setstate(std::ios::failbit);
176  break;
177  }
178  if (!std::isxdigit(c, source.getloc())) {
179  source.unget();
180  source.setstate(std::ios::failbit);
181  break;
182  }
183 
184  if (std::isdigit(c, source.getloc())) {
185  a[i] = (c - '0');
186  } else {
187  a[i] = std::toupper(c, source.getloc()) - 'A' + 0xA;
188  }
189  }
190  detail::pack<stream_endian::big_bit, 4, 8>(a, d);
191  return source;
