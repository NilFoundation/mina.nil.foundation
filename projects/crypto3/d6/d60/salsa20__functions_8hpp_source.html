49  void schedule_iv(key_schedule_type &schedule, const iv_type &iv) {
50  // XSalsa20

52  for (std::uint8_t itr = 0; itr < 4; itr++) {
53  schedule[itr + 6] = boost::endian::native_to_little(
54  make_uint_t(iv[word_bytes * itr], iv[word_bytes * itr + 1], iv[word_bytes * itr + 2],
55  iv[word_bytes * itr + 3]));
56  }
57

63  schedule[3] = hsalsa[2];
64  schedule[4] = hsalsa[3];
65  schedule[6] = boost::endian::native_to_little(

69  make_uint_t(iv[word_bytes * 5], iv[word_bytes * 5 + 1], iv[word_bytes * 5 + 2],
