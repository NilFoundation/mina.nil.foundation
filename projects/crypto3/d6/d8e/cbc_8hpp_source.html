243  inline static block_type end_message(const cipher_type &cipher, const block_type &plaintext,
244  const block_type &previous = block_type(),
245  const iv_type &iv = iv_type()) {
246  BOOST_ASSERT_MSG(buffer.size() >= offset, "Offset is sane");
247  uint8_t *buf = buffer.data() + offset;
248  const size_t sz = buffer.size() - offset;
249 
250  const size_t BS = block_size();
251 
252  if (sz < BS + 1) {
253  throw encoding_error(name() + ": insufficient data to encipher");
254  }
255 
256  if (sz % BS == 0) {
257  update(buffer, offset);
258 
259  // swap last two blocks
260  for (size_t i = 0; i != BS; ++i) {
261  std::swap(buffer[buffer.size() - BS + i], buffer[buffer.size() - 2 * BS + i]);
262  }
263  } else {
264  const size_t full_blocks = ((sz / BS) - 1) * BS;
265  const size_t final_bytes = sz - full_blocks;
266  BOOST_ASSERT_MSG(final_bytes > BS && final_bytes < 2 * BS,
267  "Left over size in expected range");
268 
269  secure_vector<uint8_t> last(buf + full_blocks, buf + full_blocks + final_bytes);
270  buffer.resize(full_blocks + offset);
271  update(buffer, offset);
272 
273  xor_buf(last.data(), state_ptr(), BS);
274  cipher().encrypt(last.data());
275 
276  for (size_t i = 0; i != final_bytes - BS; ++i) {
277  last[i] ^= last[i + BS];
278  last[i + BS] ^= last[i];
279  }
280 
281  cipher().encrypt(last.data());
282 
283  buffer += last;
284  }
285  }

314  inline static block_type process_block(const cipher_type &cipher, const block_type &plaintext,
315  const block_type &previous = block_type()) {
316  const size_t BS = block_size();
317 
318  BOOST_ASSERT_MSG(sz % BS == 0, "Input is full blocks");
319  size_t blocks = sz / BS;
320 
321  while (blocks) {
322  const size_t to_proc = std::min(BS * blocks, m_tempbuf.size());
323 
324  cipher().decrypt_n(buf, m_tempbuf.data(), to_proc / BS);
325 
326  xor_buf(m_tempbuf.data(), state_ptr(), BS);
327  xor_buf(&m_tempbuf[BS], buf, to_proc - BS);
328  copy_mem(state_ptr(), buf + (to_proc - BS), BS);
329 
330  copy_mem(buf, m_tempbuf.data(), to_proc);
331 
332  buf += to_proc;
333  blocks -= to_proc / BS;
334  }
335 
336  return sz;
337  }

340  const iv_type &iv = iv_type()) {
341  BOOST_ASSERT_MSG(buffer.size() >= offset, "Offset is sane");
342  const size_t sz = buffer.size() - offset;
343 
344  const size_t BS = block_size();
345

350  update(buffer, offset);
351 
352  const size_t pad_bytes = BS - padding().unpad(&buffer[buffer.size() - BS], BS);
353  buffer.resize(buffer.size() - pad_bytes); // remove padding
