74  static void S2V(const mac_type &mac, const uint8_t *text, size_t text_len) {
75  using namespace nil::crypto3::detail;
76 
77  const std::vector<uint8_t> zeros(block_size());
78 
79  secure_vector<uint8_t> V = mac->process(zeros.data(), zeros.size());
80 
81  for (size_t i = 0; i != m_ad_macs.size(); ++i) {
82  poly_double_n(V.data(), V.size());
83  V ^= m_ad_macs[i];
84  }
85 
86  if (m_nonce.size()) {
87  poly_double_n(V.data(), V.size());
88  V ^= m_nonce;
89  }
90 
91  if (text_len < block_size()) {
92  poly_double_n(V.data(), V.size());
93  xor_buf(V.data(), text, text_len);
94  V[text_len] ^= 0x80;
95  return mac->process(V);
96  }
97 
98  mac->update(text, text_len - block_size());
99  xor_buf(V.data(), &text[text_len - block_size()], block_size());
100  mac->update(V);
101 
102  return mac->final();
103  }
