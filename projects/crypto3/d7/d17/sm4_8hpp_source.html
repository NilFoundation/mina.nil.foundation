131  void schedule_key(const key_type &key) {
132  std::array<word_type, 4> K = {0};
133  K[0] = boost::endian::native_to_big(key[0]) ^ policy_type::fk[0];
134  K[1] = boost::endian::native_to_big(key[1]) ^ policy_type::fk[1];
135  K[2] = boost::endian::native_to_big(key[2]) ^ policy_type::fk[2];
136  K[3] = boost::endian::native_to_big(key[3]) ^ policy_type::fk[3];
137 
138  for (size_t i = 0; i != 32; ++i) {
139  K[i % 4] ^=
140  policy_type::tp(K[(i + 1) % 4] ^ K[(i + 2) % 4] ^ K[(i + 3) % 4] ^ policy_type::ck[i],
141  policy_type::constants);
142  key_schedule[i] = K[i % 4];
143  }
144 
145  K.fill(0);
