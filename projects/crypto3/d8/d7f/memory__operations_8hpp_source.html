65  void secure_scrub_memory(void *ptr, size_t n) {
66 #if defined(CRYPTO3_TARGET_OS_HAS_RTLSECUREZEROMEMORY)
67  ::RtlSecureZeroMemory(ptr, n);
68 
69 #elif defined(CRYPTO3_TARGET_OS_HAS_EXPLICIT_BZERO)
70  ::explicit_bzero(ptr, n);
71 
72 #elif defined(CRYPTO3_USE_VOLATILE_MEMSET_FOR_ZERO) && (CRYPTO3_USE_VOLATILE_MEMSET_FOR_ZERO == 1)
73  /*
74  * Call memset through a static volatile pointer, which the compiler
75  * should not elide. This construct should be safe in conforming
76  * compilers, but who knows. I did confirm that on x86-64 GCC 6.1 and
77  * Clang 3.8 both create code that saves the memset address in the
78  * data segment and uncondtionally loads and jumps to that address.
79  */
80  static void *(*const volatile memset_ptr)(void *, int, size_t) = std::memset;
81  (memset_ptr)(ptr, 0, n);
82 #else
83 
84  volatile uint8_t *p = reinterpret_cast<volatile uint8_t *>(ptr);
85 
86  for (size_t i = 0; i != n; ++i) {
87  p[i] = 0;
88  }
89 #endif

233  inline void xor_buf(uint8_t out[], const uint8_t in[], size_t length) {
234  while (length >= 16) {
235  uint64_t x0, x1, y0, y1;
236  std::memcpy(&x0, in, 8);
237  std::memcpy(&x1, in + 8, 8);
238  std::memcpy(&y0, out, 8);
239  std::memcpy(&y1, out + 8, 8);
240 
241  y0 ^= x0;
242  y1 ^= x1;
243  std::memcpy(out, &y0, 8);
244  std::memcpy(out + 8, &y1, 8);
245  out += 16;
246  in += 16;
247  length -= 16;
248  }
249 
250  while (length > 0) {
251  out[0] ^= in[0];
252  out += 1;
253  in += 1;
254  length -= 1;
255  }
256  }
