62  static void poly1305_blocks(key_schedule_type &X, const uint8_t *m, size_t blocks,
63  bool is_final = false) {
64  const word_type hibit = is_final ? 0 : (static_cast<word_type>(1) << 40); /* 1 << 128 */
65 
66  const word_type r0 = X[0];
67  const word_type r1 = X[1];
68  const word_type r2 = X[2];
69 
70  word_type h0 = X[3 + 0];
71  word_type h1 = X[3 + 1];
72  word_type h2 = X[3 + 2];
73 
74  const word_type s1 = r1 * (5 << 2);
75  const word_type s2 = r2 * (5 << 2);
76 
77  while (blocks--) {
78  /* h += m[i] */
79  const word_type t0 = load_le<word_type>(m, 0);
80  const word_type t1 = load_le<word_type>(m, 1);
81 
82  h0 += ((t0)&0xfffffffffff);
83  h1 += (((t0 >> 44) | (t1 << 20)) & 0xfffffffffff);
84  h2 += (((t1 >> 24)) & 0x3ffffffffff) | hibit;
85 
86  /* h *= r */
87  uint128_t d0 = uint128_t(h0) * r0 + uint128_t(h1) * s2 + uint128_t(h2) * s1;
88  uint128_t d1 = uint128_t(h0) * r1 + uint128_t(h1) * r0 + uint128_t(h2) * s2;
89  uint128_t d2 = uint128_t(h0) * r2 + uint128_t(h1) * r1 + uint128_t(h2) * r0;
90 
91  /* (partial) h %= p */
92  word_type c = carry_shift(d0, 44);
93  h0 = d0 & 0xfffffffffff;
94  d1 += c;
95  c = carry_shift(d1, 44);
96  h1 = d1 & 0xfffffffffff;
97  d2 += c;
98  c = carry_shift(d2, 42);
99  h2 = d2 & 0x3ffffffffff;
100  h0 += c * 5;
101  c = carry_shift(h0, 44);
102  h0 = h0 & 0xfffffffffff;
103  h1 += c;
104 
105  m += 16;
106  }
107 
108  X[3 + 0] = h0;
109  X[3 + 1] = h1;
110  X[3 + 2] = h2;
111  }

113  static void poly1305_finish(key_schedule_type &X, uint8_t mac[16]) {
114  /* fully carry h */
115  word_type h0 = X[3 + 0];
116  word_type h1 = X[3 + 1];
117  word_type h2 = X[3 + 2];
118 
119  word_type c;
120  c = (h1 >> 44);
121  h1 &= 0xfffffffffff;
122  h2 += c;
123  c = (h2 >> 42);
124  h2 &= 0x3ffffffffff;
125  h0 += c * 5;
126  c = (h0 >> 44);
127  h0 &= 0xfffffffffff;
128  h1 += c;
129  c = (h1 >> 44);
130  h1 &= 0xfffffffffff;
131  h2 += c;
132  c = (h2 >> 42);
133  h2 &= 0x3ffffffffff;
134  h0 += c * 5;
135  c = (h0 >> 44);
136  h0 &= 0xfffffffffff;
137  h1 += c;
138 
139  /* compute h + -p */
140  word_type g0 = h0 + 5;
141  c = (g0 >> 44);
142  g0 &= 0xfffffffffff;
143  word_type g1 = h1 + c;
144  c = (g1 >> 44);
145  g1 &= 0xfffffffffff;
146  word_type g2 = h2 + c - (static_cast<word_type>(1) << 42);
147 
148  /* select h if h < p, or h + -p if h >= p */
149  c = (g2 >> ((sizeof(word_type) * 8) - 1)) - 1;
150  g0 &= c;
151  g1 &= c;
152  g2 &= c;
153  c = ~c;
154  h0 = (h0 & c) | g0;
155  h1 = (h1 & c) | g1;
156  h2 = (h2 & c) | g2;
157 
158  /* h = (h + pad) */
159  const word_type t0 = X[6];
160  const word_type t1 = X[7];
161 
162  h0 += ((t0)&0xfffffffffff);
163  c = (h0 >> 44);
164  h0 &= 0xfffffffffff;
165  h1 += (((t0 >> 44) | (t1 << 20)) & 0xfffffffffff) + c;
166  c = (h1 >> 44);
167  h1 &= 0xfffffffffff;
168  h2 += (((t1 >> 24)) & 0x3ffffffffff) + c;
169  h2 &= 0x3ffffffffff;
170 
171  /* mac = h % (2^128) */
172  h0 = ((h0) | (h1 << 44));
173  h1 = ((h1 >> 20) | (h2 << 24));
174 
175  store_le(mac, h0, h1);
176 
177  /* zero out the state */
178  clear_mem(X.data(), X.size());
179  }
