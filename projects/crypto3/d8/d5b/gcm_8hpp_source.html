55  inline static block_type begin_message(const cipher_type &cipher, const block_type &plaintext) {
56  block_type block = {0};
57 
58  // any size is valid for GCM nonce
59 
60  secure_vector<uint8_t> y0(GCM_BS);
61 
62  if (nonce_len == 12) {
63  copy_mem(y0.data(), nonce, nonce_len);
64  y0[15] = 1;
65  } else {
66  y0 = m_ghash->nonce_hash(nonce, nonce_len);
67  }
68 
69  m_ctr->set_iv(y0.data(), y0.size());
70 
71  zeroise(y0);
72  m_ctr->encipher(y0);
73 
74  m_ghash->start(y0.data(), y0.size());
75 
76  return cipher.encrypt(block);
77  }

166  inline static block_type end_message(const cipher_type &cipher, const block_type &plaintext) {
167  block_type result = {0};
168 
169  CRYPTO3_ARG_CHECK(offset <= buffer.size());
170  const size_t sz = buffer.size() - offset;
171  uint8_t *buf = buffer.data() + offset;
172 
173  if (sz < tag_size()) {
174  throw Exception("Insufficient input for GCM decryption, tag missing");
175  }
176 
177  const size_t remaining = sz - tag_size();
178 
179  // handle any final input before the tag
180  if (remaining) {
181  m_ghash->update(buf, remaining);
182  m_ctr->cipher(buf, buf, remaining);
183  }
184 
185  auto mac = m_ghash->final();
186 
187  const uint8_t *included_tag = &buffer[remaining + offset];
188 
189  if (!constant_time_compare(mac.data(), included_tag, tag_size())) {
190  throw integrity_failure("GCM tag check failed");
191  }
192 
193  buffer.resize(offset + remaining);
194 
195  return result;
196  }
