133  inline digest_type digest() const {
134  word_type x = (state_[0] << (DigestBits / 2)) | state_[1];
135  digest_type d;
136  // RFC 1950, Section 2.2 stores the ADLER-32 in big-endian
137  ::nil::crypto3::detail::pack_n<stream_endian::big_bit, digest_bits, octet_bits>(
138  &x, 1, d.data(), digest_bits / octet_bits);
139  return d;
140  }

142  inline digest_type end_message() {
143  digest_type d(std::move(digest()));
144  reset();
145  return d;

150  if (DigestBits < 16) {
151  x %= modulo;

155 #endif

157  state_[0] = (state_[0] + state_[1]) % modulo;

164  template<typename InputIterator>
165  inline basic_adler &update_n(InputIterator p, size_t n) {
166 #ifndef CRYPTO3_HASH_NO_OPTIMIZATION
167 
168  unsigned const fast_word_bits = (word_bits < 16 ? 16 : word_bits);
169  typedef typename boost::uint_t<fast_word_bits>::least /*fast*/ fast_word_type;
170  /*
171 
172  Worst-case behaviour for delaying the modulo:
173  - every input is 255
174  - s1 and s0 start out at modulo-1
175 
176  So after k inputs, we have:
177  - s1 = (modulo-1) + k*255
178  - s0 = (modulo-1) + Sigma(i = 1 to k)[ (modulo-1) + i*255 ]
179  = (modulo-1) + k*(modulo-1) + Sigma(i = 1 to k)[ i*255 ]
180  = (k+1)*(modulo-1) + 255 * Sigma(i = 1 to k)[i]
181  = (k+1)*(modulo-1) + 255 * k*(k+1)/2
182 
183  And to avoid overflow we need s1, s0 <= 2**fast_word_bits - 1
184 
185  s1 = (modulo-1) + k*255 <= 2**fast_word_bits - 1
186  k*255 <= 2**fast_word_bits - 1 - (modulo-1)
187  k <= (2**fast_word_bits - modulo)/255
188 
189  Then use an overestimate for s0 to make the numbers nicer
190  s0 < (k+1)*modulo + 256/2(k+1)**2 < 2**fast_word_bits
191 
192  Which solves as
193  k < ( sqrt(512*2**fast_word_bits + modulo**2) - m - 256 )/256
194 
195  So then overestimating m as 2**(word_bits/2) and other safe approximations gives
196  k < 2**((fast_word_bits-7)/2) - 2**((word_bits-16)/2) - 1
197 
198  Bits Limit
199  ---- -----
200  8 16
201  16 16
202  24 240
203  32 3840
204  40 61440
205  48 983040
206  56 15728640
207  64 251658240
208 
209  */
210 
211  unsigned const less = (1 << (fast_word_bits / 2 - 8));
212  unsigned const limit = (1 << (fast_word_bits / 2 - 4)) - (word_bits < 16 ? 0 : less);
213 
214 #define CRYPTO3_HASH_ADLER_STEP \
215  { \
216  value_type x = *p++; \
217  s1 += x; \
218  s0 += s1; \
219  }
220 
221 #define CRYPTO3_HASH_ADLER_8_STEPS \
222  {CRYPTO3_HASH_ADLER_STEP CRYPTO3_HASH_ADLER_STEP CRYPTO3_HASH_ADLER_STEP CRYPTO3_HASH_ADLER_STEP \
223  CRYPTO3_HASH_ADLER_STEP CRYPTO3_HASH_ADLER_STEP CRYPTO3_HASH_ADLER_STEP CRYPTO3_HASH_ADLER_STEP}
224 
225  fast_word_type s0 = state_[0];
226  fast_word_type s1 = state_[1];
227 
228  for (; n >= limit; n -= limit) {
229  unsigned m = limit;
230  for (; m >= 8; m -= 8) {
231  CRYPTO3_HASH_ADLER_8_STEPS
232  }
233  while (m--) {
234  CRYPTO3_HASH_ADLER_STEP
235  }
236  s1 %= modulo;
237  s0 %= modulo;
238  }
239  for (; n >= 8; n -= 8) {
240  CRYPTO3_HASH_ADLER_8_STEPS
241  }
242  while (n--) {
243  CRYPTO3_HASH_ADLER_STEP
244  }
245  s1 %= modulo;
246  s0 %= modulo;
247 
248  state_[0] = s0;
249  state_[1] = s1;
250 
251 #else
252  while (n--)
253  update_one(*p++);
254 #endif
255  return *this;
256  }
