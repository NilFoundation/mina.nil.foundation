214  inline static block_type end_message(const cipher_type &cipher, const block_type &plaintext) {
215  block_type result = {0};
216 
217  BOOST_ASSERT_MSG(buffer.size() >= offset, "Offset is sane");
218  const size_t sz = buffer.size() - offset;
219  uint8_t *buf = buffer.data() + offset;
220 
221  BOOST_ASSERT_MSG(sz >= tag_size(), "Have the tag as part of final input");
222 
223  const size_t remaining = sz - tag_size();
224 
225  if (remaining) {
226  m_poly1305->update(buf, remaining); // poly1305 of ciphertext
227  m_chacha->cipher1(buf, remaining);
228  m_ctext_len += remaining;
229  }
230 
231  if (cfrg_version()) {
232  if (m_ctext_len % 16) {
233  const uint8_t zeros[16] = {0};
234  m_poly1305->update(zeros, 16 - m_ctext_len % 16);
235  }
236  update_len(m_ad.size());
237  }
238 
239  update_len(m_ctext_len);
240  const secure_vector<uint8_t> mac = m_poly1305->final();
241 
242  const uint8_t *included_tag = &buf[remaining];
243 
244  m_ctext_len = 0;
245 
246  if (!constant_time_compare(mac.data(), included_tag, tag_size())) {
247  throw integrity_failure("ChaCha20Poly1305 tag check failed");
248  }
249  buffer.resize(offset + remaining);
250 
251  return result;
252  }
