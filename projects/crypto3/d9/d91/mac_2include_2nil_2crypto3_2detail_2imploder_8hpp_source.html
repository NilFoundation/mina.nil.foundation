94  struct imploder<Endian<UnitBits>, InputBits, OutputBits, k> {
95 
96  // To keep the implementation managable, input and output sizes must
97  // be multiples or factors of the unit size.
98  // If one of these is firing, you may want a bit-only stream_endian
99  // rather than one that mentions bytes or octets.
100  BOOST_STATIC_ASSERT(!(InputBits % UnitBits && UnitBits % InputBits));
101  BOOST_STATIC_ASSERT(!(OutputBits % UnitBits && UnitBits % OutputBits));
102 
103  typedef Endian<UnitBits> Endianness;
104  typedef imploder_step<Endianness, InputBits, OutputBits, k> step_type;
105  typedef imploder<Endianness, InputBits, OutputBits, k + InputBits> next_type;

107  template<typename InIter, typename OutputValue>
108  static void implode(InIter &in, OutputValue &x) {
109  step_type::step(*in++, x);
