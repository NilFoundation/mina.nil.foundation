159  inline void schedule_key(const key_type &key) {
160  std::array<word_type, 4> WK = {0};
161 
162  for (size_t i = 0; i != 4; ++i) {
163  WK[i] = boost::endian::native_to_big(key[i]);
164  }
165 
166  for (size_t i = 0; i != 16; i += 2) {
167  key_schedule[2 * i] = policy_type::g(WK[0] + WK[2] - policy_type::round_constants[i],
168  policy_type::s0_constants, policy_type::s1_constants,
169  policy_type::s2_constants, policy_type::s3_constants);
170  key_schedule[2 * i + 1] = policy_type::g(WK[1] - WK[3] + policy_type::round_constants[i],
171  policy_type::s0_constants, policy_type::s1_constants,
172  policy_type::s2_constants, policy_type::s3_constants) ^
173  key_schedule[2 * i];
174 
175  uint32_t T = (WK[0] & 0xFF) << 24;
176  WK[0] = (WK[0] >> 8) | (::nil::crypto3::detail::extract_uint_t<CHAR_BIT>(WK[1], 3) << 24);
177  WK[1] = (WK[1] >> 8) | T;
178 
179  key_schedule[2 * i + 2] = policy_type::g(WK[0] + WK[2] - policy_type::round_constants[i + 1],
180  policy_type::s0_constants, policy_type::s1_constants,
181  policy_type::s2_constants, policy_type::s3_constants);
182  key_schedule[2 * i + 3] = policy_type::g(WK[1] - WK[3] + policy_type::round_constants[i + 1],
183  policy_type::s0_constants, policy_type::s1_constants,
184  policy_type::s2_constants, policy_type::s3_constants) ^
185  key_schedule[2 * i + 2];
186 
187  T = ::nil::crypto3::detail::extract_uint_t<CHAR_BIT>(WK[3], 0);
188  WK[3] = (WK[3] << 8) | ::nil::crypto3::detail::extract_uint_t<CHAR_BIT>(WK[2], 0);
189  WK[2] = (WK[2] << 8) | T;
190  }
191  }
