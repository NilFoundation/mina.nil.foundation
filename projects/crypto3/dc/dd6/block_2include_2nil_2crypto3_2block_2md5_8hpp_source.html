100

102  for (unsigned t = 0; t < block_words; ++t) {
103  std::printf("%c%c = %.8x\n", 'A' + t, 'A' + t, plaintext[t]);

107  // Initialize working variables with block
108  word_type a = plaintext[0], b = plaintext[1], c = plaintext[2], d = plaintext[3];
109 
110  // Encipher block
111 #define CRYPTO3_BLOCK_MD5_ENCRYPT_STEP(aa, bb, cc, dd, fun, k, s, i) \
112  { \
113  word_type T = \
114  aa + policy_type::fun(bb, cc, dd) + key[policy_type::key_indexes[k]] + policy_type::constants[i - 1]; \
115  aa = bb + policy_type::rotl<s>(T); \
116  }

118  CRYPTO3_BLOCK_MD5_ENCRYPT_STEP(a, b, c, d, ff, t + 0, 7, t + 1)

162 
163 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS

169  // Initialize working variables with block

172  // Decipher block
173 #define CRYPTO3_BLOCK_MD5_DECRYPT_STEP(aa, bb, cc, dd, fun, k, s, i) \
174  { \
175  word_type T = policy_type::rotr<s>(aa - bb); \
