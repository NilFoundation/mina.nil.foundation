37  static void update_tweak(const cipher_type &tweak_cipher, size_t which) {
38  const size_t BS = tweak_cipher->block_size();
39 
40  if (which > 0) {
41  poly_double_n_le(m_tweak.data(), &m_tweak[(which - 1) * BS], BS);
42  }
43 
44  const size_t blocks_in_tweak = update_granularity() / BS;
45 
46  for (size_t i = 1; i < blocks_in_tweak; ++i) {
47  poly_double_n_le(&m_tweak[i * BS], &m_tweak[(i - 1) * BS], BS);
48  }
49  }

51  static void schedule_key(const key_type &key) {
52  const size_t key_half = length / 2;
53 
54  if (length % 2 == 1 || !m_cipher->valid_keylength(key_half)) {
55  throw invalid_key_length(name(), length);
56  }
57 
58  m_cipher->set_key(key, key_half);

78  block_type begin_message(const cipher_type &cipher, const block_type &plaintext) {
79  if (!valid_nonce_length(nonce_len)) {
80  throw invalid_iv_length(name(), nonce_len);
81  }
82 
83  copy_mem(m_tweak.data(), nonce, nonce_len);
84  m_tweak_cipher->encrypt(m_tweak.data());
85 
86  update_tweak(0);
87  }
88 
89  block_type process_block(const cipher_type &cipher, const block_type &plaintext) {
90  const size_t BS = cipher().block_size();
91 
92  BOOST_ASSERT_MSG(sz % BS == 0, "Input is full blocks");
93  size_t blocks = sz / BS;
94 
95  const size_t blocks_in_tweak = update_granularity() / BS;
96 
97  while (blocks) {
98  const size_t to_proc = std::min(blocks, blocks_in_tweak);
99 
100  cipher().encrypt_n_xex(buf, tweak(), to_proc);
101 
102  buf += to_proc * BS;
103  blocks -= to_proc;
104 
105  update_tweak(to_proc);
106  }
107 
108  return sz;
109  }
110 
111  block_type end_message(const cipher_type &cipher, const block_type &plaintext) {
112  BOOST_ASSERT_MSG(buffer.size() >= offset, "Offset is sane");
113  const size_t sz = buffer.size() - offset;
114  uint8_t *buf = buffer.data() + offset;
115 
116  BOOST_ASSERT_MSG(sz >= minimum_final_size(), "Have sufficient final input in XTS encipher");
117 
118  const size_t BS = cipher().block_size();
119 
120  if (sz % BS == 0) {
121  update(buffer, offset);
122  } else {
123  // steal ciphertext
124  const size_t full_blocks = ((sz / BS) - 1) * BS;
125  const size_t final_bytes = sz - full_blocks;
126  BOOST_ASSERT_MSG(final_bytes > BS && final_bytes < 2 * BS,
127  "Left over size in expected range");
128 
129  secure_vector<uint8_t> last(buf + full_blocks, buf + full_blocks + final_bytes);
130  buffer.resize(full_blocks + offset);
131  update(buffer, offset);
132 
133  xor_buf(last, tweak(), BS);
134  cipher().encrypt(last);
135  xor_buf(last, tweak(), BS);
136 
137  for (size_t i = 0; i != final_bytes - BS; ++i) {
138  last[i] ^= last[i + BS];
139  last[i + BS] ^= last[i];
140  last[i] ^= last[i + BS];
141  }
142 
143  xor_buf(last, tweak() + BS, BS);
144  cipher().encrypt(last);
145  xor_buf(last, tweak() + BS, BS);
146 
147  buffer += last;
148  }
149  }
