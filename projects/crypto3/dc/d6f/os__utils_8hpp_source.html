79  uint64_t CRYPTO3_TEST_API get_processor_timestamp() {
80  uint64_t rtc = 0;
81 
82 #if defined(CRYPTO3_TARGET_OS_HAS_WIN32)
83  LARGE_INTEGER tv;
84  ::QueryPerformanceCounter(&tv);
85  rtc = tv.QuadPart;
86 
87 #elif defined(CRYPTO3_USE_GCC_INLINE_ASM)
88 
89 #if defined(CRYPTO3_TARGET_CPU_IS_X86_FAMILY)
90 
91  if (cpuid::has_rdtsc()) {
92  uint32_t rtc_low = 0, rtc_high = 0;
93  asm volatile("rdtsc" : "=d"(rtc_high), "=a"(rtc_low));
94  rtc = (static_cast<uint64_t>(rtc_high) << 32) | rtc_low;
95  }
96 
97 #elif defined(CRYPTO3_TARGET_ARCHITECTURE_IS_PPC64)
98 
99  for (;;) {
100  uint32_t rtc_low = 0, rtc_high = 0, rtc_high2 = 0;
101  asm volatile("mftbu %0" : "=r"(rtc_high));
102  asm volatile("mftb %0" : "=r"(rtc_low));
103  asm volatile("mftbu %0" : "=r"(rtc_high2));
104 
105  if (rtc_high == rtc_high2) {
106  rtc = (static_cast<uint64_t>(rtc_high) << 32) | rtc_low;
107  break;
108  }
109  }
110 
111 #elif defined(CRYPTO3_TARGET_ARCHITECTURE_IS_ALPHA)
112  asm volatile("rpcc %0" : "=r"(rtc));
113 
114  // OpenBSD does not trap access to the %tick register
115 #elif defined(CRYPTO3_TARGET_ARCHITECTURE_IS_SPARC64) && !defined(CRYPTO3_TARGET_OS_IS_OPENBSD)
116  asm volatile("rd %%tick, %0" : "=r"(rtc));
117 
118 #elif defined(CRYPTO3_TARGET_ARCHITECTURE_IS_IA64)
119  asm volatile("mov %0=ar.itc" : "=r"(rtc));
120 
121 #elif defined(CRYPTO3_TARGET_ARCHITECTURE_IS_S390X)
122  asm volatile("stck 0(%0)" : : "a"(&rtc) : "memory", "cc");
123 
124 #elif defined(CRYPTO3_TARGET_ARCHITECTURE_IS_HPPA)
125  asm volatile("mfctl 16,%0" : "=r"(rtc)); // 64-bit only?
126 
127 #else
128  //#warning "get_processor_timestamp not implemented"
129 #endif
130 
131 #endif

213  size_t get_memory_locking_limit() {
214 #if defined(CRYPTO3_TARGET_OS_HAS_POSIX1)
215  /*
216  * Linux defaults to only 64 KiB of mlockable memory per process
217  * (too small) but BSDs offer a small fraction of total RAM (more
218  * than we need). Bound the total mlock size to 512 KiB which is
219  * enough to run the entire test suite without spilling to non-mlock
220  * memory (and thus presumably also enough for many useful
221  * programs), but small enough that we should not cause problems
222  * even if many processes are mlocking on the same machine.
223  */
224  size_t mlock_requested = CRYPTO3_MLOCK_ALLOCATOR_MAX_LOCKED_KB;
225 
226  /*
227  * Allow override via env variable
228  */
229  if (const char *env = std::getenv("CRYPTO3_MLOCK_POOL_SIZE")) {
230  try {
231  const size_t user_req = std::stoul(env, nullptr);
232  mlock_requested = std::min(user_req, mlock_requested);
233  } catch (std::exception &) { /* ignore it */
234  }
235  }
236 
237 #if defined(RLIMIT_MEMLOCK)
238  if (mlock_requested > 0) {
239  struct ::rlimit limits;
240 
241  ::getrlimit(RLIMIT_MEMLOCK, &limits);
242 
243  if (limits.rlim_cur < limits.rlim_max) {
244  limits.rlim_cur = limits.rlim_max;
245  ::setrlimit(RLIMIT_MEMLOCK, &limits);
246  ::getrlimit(RLIMIT_MEMLOCK, &limits);
247  }
248 
249  return std::min<size_t>(limits.rlim_cur, mlock_requested * 1024);
250  }
251 #else
252  /*
253  * If RLIMIT_MEMLOCK is not defined, likely the OS does not support
254  * unprivileged mlock calls.
255  */
256  return 0;
257 #endif
258 
259 #elif defined(CRYPTO3_TARGET_OS_HAS_VIRTUAL_LOCK)
260  SIZE_T working_min = 0, working_max = 0;
261  if (!::GetProcessWorkingSetSize(::GetCurrentProcess(), &working_min, &working_max)) {
262  return 0;
263  }
264 
265  // According to Microsoft MSDN:
266  // The maximum number of pages that a process can lock is equal to the number of pages in its minimum
267  // working set minus a small overhead In the book "Windows Internals Part 2": the maximum lockable pages
268  // are minimum working set size - 8 pages But the information in the book seems to be
269  // inaccurate/outdated I've tested this on Windows 8.1 x64, Windows 10 x64 and Windows 7 x86 On all
270  // three OS the value is 11 instead of 8
271  size_t overhead = system_page_size() * 11ULL;
272  if (working_min > overhead) {
273  size_t lockable_bytes = working_min - overhead;
274  if (lockable_bytes < (CRYPTO3_MLOCK_ALLOCATOR_MAX_LOCKED_KB * 1024ULL)) {
275  return lockable_bytes;
276  } else {
277  return CRYPTO3_MLOCK_ALLOCATOR_MAX_LOCKED_KB * 1024ULL;
278  }
279  }
280 #endif
281 
282  return 0;
283  }

413  int CRYPTO3_TEST_API run_cpu_instruction_probe(std::function<int()> probe_fn) {
414  volatile int probe_result = -3;
415 
416 #if defined(CRYPTO3_TARGET_OS_HAS_POSIX1)
417  struct sigaction old_sigaction;
418  struct sigaction sigaction;
419 
420  sigaction.sa_handler = detail::sigill_handler;
421  sigemptyset(&sigaction.sa_mask);
422  sigaction.sa_flags = 0;
423 
424  int rc = ::sigaction(SIGILL, &sigaction, &old_sigaction);
425 
426  if (rc != 0) {
427  throw Exception("run_cpu_instruction_probe sigaction failed");
428  }
429 
430  rc = sigsetjmp(detail::sigill_jmp_buf, /*save sigs*/ 1);
431 
432  if (rc == 0) {
433  // first call to sigsetjmp
434  probe_result = probe_fn();
435  } else if (rc == 1) {
436  // non-local return from siglongjmp in signal handler: return error
437  probe_result = -1;
438  }
439 
440  // Restore old SIGILL handler, if any
441  rc = ::sigaction(SIGILL, &old_sigaction, nullptr);
442  if (rc != 0) {
443  throw Exception("run_cpu_instruction_probe sigaction restore failed");
444  }
445 
446 #elif defined(CRYPTO3_TARGET_OS_IS_WINDOWS) && defined(CRYPTO3_TARGET_COMPILER_IS_MSVC)
447 
448  // Windows SEH
449  __try {
450  probe_result = probe_fn();
451  } __except (::GetExceptionCode() == EXCEPTION_ILLEGAL_INSTRUCTION ? EXCEPTION_EXECUTE_HANDLER :
452  EXCEPTION_CONTINUE_SEARCH) {
453  probe_result = -1;
454  }
455 
456 #endif
457 
458  return probe_result;
459  }
