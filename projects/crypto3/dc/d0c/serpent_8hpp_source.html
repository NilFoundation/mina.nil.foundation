306  void schedule_key(const key_type &key) {
307  std::array<word_type, 140> W = {0};
308  for (size_t i = 0; i != key.size() / 4; ++i) {
309  W[i] = boost::endian::native_to_little(key[i]);
310  }
311 
312  W[key.size() / 4] |= word_type(1) << ((key.size() % 4) * 8);
313 
314  for (size_t i = 8; i != 140; ++i) {
315  word_type wi = W[i - 8] ^ W[i - 5] ^ W[i - 3] ^ W[i - 1] ^ policy_type::phi ^ word_type(i - 8);
316  W[i] = policy_type::template rotl<11>(wi);
317  }
