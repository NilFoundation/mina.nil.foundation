87  template<typename Dummy>
88  typename boost::enable_if_c<length_bits && sizeof(Dummy)>::type append_length(length_type length) {
89  using namespace nil::crypto3::detail;

92  // Convert the input into words

97  std::array<length_type, 1> length_array = {{length}};

99  pack<endian_type, length_bits, word_bits>(length_array, length_words_array);

101  block[block_words - i] = length_words_array[length_words - i];

114  merkle_damgard_stream_processor &update_one(value_type value) {
115  using namespace boost::accumulators;
116  length_type seen = accumulators::extract::bits_count(acc);
117  std::size_t i = seen % block_bits;
118  std::size_t j = i / value_bits;
119  value_array[cache_size] = value;
120  ++cache_size;
121  if (cache_size == block_values) {
122  // Process the completed block
123  process_block();
124  cache_size = 0;

129  template<typename InputIterator>
130  merkle_damgard_stream_processor &update_n(InputIterator p, size_t n) {
131  using namespace nil::crypto3::detail;
132  using namespace boost::accumulators;
133 #ifndef CRYPTO3_HASH_NO_OPTIMIZATION
134  length_type seen = accumulators::extract::bits_count(acc);
135  if (seen % block_bits != 0) {
136  for (; n && (block_bits - seen % block_bits); --n, ++p) {
137  value_array[cache_size] = *p;
138  ++cache_size;
139  }
140  process_block(cache_size * value_bits);
141  cache_size = 0;
142  }
143  for (; n >= block_values; n -= block_values, p += block_values) {
144  // Convert the input into words
145  block_type block;
146  pack_n<endian_type, value_bits, word_bits>(p, block_values, std::begin(block), block_words);
147 
148  // Process the block
149  std::size_t bb = block_bits;
150  acc(block, accumulators::bits = bb);
151  }
152 #endif
153 
154  for (; n; --n, ++p) {
155  value_array[cache_size] = *p;
156  ++cache_size;
157  }
158  process_block(cache_size * value_bits);
159  cache_size = 0;

188  digest_type end_message() {
189  using namespace nil::crypto3::detail;
190  using namespace boost::accumulators;
191 
192  length_type seen = count(acc), length = seen;
193 
194  // Add a 1 bit
195 #ifdef CRYPTO3_HASH_NO_OPTIMIZATION
196  std::array<bool, value_bits> padding_bits = {{1}};
197  std::array<value_type, 1> padding_values;
198  pack<endian_type, 1, value_bits>(padding_bits, padding_values);
199  update_one(padding_values[0]);
200 #else
201  value_type pad = 0;
202  imploder_step<endian_type, 1, value_bits, 0>::step(1, pad);
203  update_one(pad);
204 #endif
205 
206  // Pad with 0 bits
207  while ((seen + length_bits) % block_bits != 0) {
208  update_one(value_type());
209  }
210 
211  // Append length
212  append_length<int>(length);
213 
214  cache_size = 0;
215  // Calculate static_digest and reset block_hash
216  return block_hash.end_message();
217  }
