161  inline static block_type end_message(const cipher_type &cipher, const block_type &plaintext) {
162  block_type result = {0};
163  BOOST_ASSERT_MSG(buffer.size() >= offset, "Offset is sane");
164  const size_t sz = buffer.size() - offset;
165  uint8_t *buf = buffer.data() + offset;
166 
167  BOOST_ASSERT_MSG(sz >= tag_size(), "Have the tag as part of final input");
168 
169  const size_t remaining = sz - tag_size();
170 
171  if (remaining) {
172  m_cmac->update(buf, remaining);
173  m_ctr->cipher(buf, buf, remaining);
174  }
175 
176  const uint8_t *included_tag = &buf[remaining];
177 
178  secure_vector<uint8_t> mac = m_cmac->final();
179  mac ^= m_nonce_mac;
180 
181  if (m_ad_mac.empty()) {
182  m_ad_mac = eax_prf(1, block_size(), *m_cmac, nullptr, 0);
183  }
184 
185  mac ^= m_ad_mac;
186 
187  if (!constant_time_compare(mac.data(), included_tag, tag_size())) {
188  throw integrity_failure("EAX tag check failed");
189  }
190 
191  buffer.resize(offset + remaining);
192  return result;
193  }
