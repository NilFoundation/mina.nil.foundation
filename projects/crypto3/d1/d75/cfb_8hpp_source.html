36  static void shift_register(cipher_type &cipher) {
37  const size_t shift = feedback();
38  const size_t carryover = block_size() - shift;
39 
40  if (carryover > 0) {
41  copy_mem(m_state.data(), &m_state[shift], carryover);
42  }
43  copy_mem(&m_state[carryover], m_keystream.data(), shift);
44  cipher().encrypt(m_state, m_keystream);
45  m_keystream_pos = 0;
46  }

48  static inline void xor_copy(uint8_t buf[], uint8_t key_buf[], size_t len) {
49  for (size_t i = 0; i != len; ++i) {
50  uint8_t k = key_buf[i];
51  key_buf[i] = buf[i];
52  buf[i] ^= k;
53  }
54  }
55  };

73  inline static block_type begin_message(const cipher_type &cipher, const block_type &plaintext) {
74  if (!valid_nonce_length(nonce_len)) {
75  throw invalid_iv_length(name(), nonce_len);
76  }
77 
78  if (nonce_len == 0) {
79  if (m_state.empty()) {
80  throw invalid_state("CFB requires a non-empty initial nonce");
81  }
82  // No reason to encipher state->keystream_buf, because no change
83  } else {
84  m_state.assign(nonce, nonce + nonce_len);
85  m_keystream.resize(m_state.size());
86  cipher().encrypt(m_state, m_keystream);
87  m_keystream_pos = 0;
88  }
89  }

91  inline static block_type process_block(const cipher_type &cipher, const block_type &plaintext) {
92  const size_t shift = feedback();
93 
94  size_t left = sz;
95 
96  if (m_keystream_pos != 0) {
97  const size_t take = std::min<size_t>(left, shift - m_keystream_pos);
98 
99  xor_buf(m_keystream.data() + m_keystream_pos, buf, take);
100  copy_mem(buf, m_keystream.data() + m_keystream_pos, take);
101 
102  m_keystream_pos += take;
103  left -= take;
104  buf += take;
105 
106  if (m_keystream_pos == shift) {
107  shift_register();
108  }
109  }
110 
111  while (left >= shift) {
112  xor_buf(m_keystream.data(), buf, shift);
113  copy_mem(buf, m_keystream.data(), shift);
114 
115  left -= shift;
116  buf += shift;
117  shift_register();
118  }
119 
120  if (left > 0) {
121  xor_buf(m_keystream.data(), buf, left);
122  copy_mem(buf, m_keystream.data(), left);
123  m_keystream_pos += left;
124  }
125 
126  return sz;
127  }
128 
129  inline static block_type end_message(const cipher_type &cipher, const block_type &plaintext) {
130  block_type result = {0};
131 
132  return result;
133  }
