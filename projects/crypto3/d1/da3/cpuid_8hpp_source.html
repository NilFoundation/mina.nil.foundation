429  static endian_status runtime_check_endian() {
430  // Check runtime endian
431  const uint32_t endian32 = 0x01234567;
432  const uint8_t *e8 = reinterpret_cast<const uint8_t *>(&endian32);
433 
434  endian_status endian = ENDIAN_UNKNOWN;
435 
436  if (e8[0] == 0x01 && e8[1] == 0x23 && e8[2] == 0x45 && e8[3] == 0x67) {
437  endian = ENDIAN_BIG;
438  } else if (e8[0] == 0x67 && e8[1] == 0x45 && e8[2] == 0x23 && e8[3] == 0x01) {
439  endian = ENDIAN_LITTLE;
440  } else {
441  throw std::exception();
442  }
443 
444  // If we were compiled with a known endian, verify it matches at runtime
445 #if defined(CRYPTO3_TARGET_CPU_IS_LITTLE_ENDIAN)
446  BOOST_ASSERT_MSG(endian == ENDIAN_LITTLE, "Build and runtime endian match");
447 #elif defined(CRYPTO3_TARGET_CPU_IS_BIG_ENDIAN)
448  BOOST_ASSERT_MSG(endian == ENDIAN_BIG, "Build and runtime endian match");
449 #endif
