94  template<typename OutputIterator>
95  OutputIterator update_one(value_type value, OutputIterator out) {
96  unsigned i = seen % block_bits;
97  unsigned j = i / value_bits;
98  value_array[j] = value;
99  seen += value_bits;
100  if (i == block_bits - value_bits) {
101  // Process the completed block
102  // Convert the input into words
103  block_type block = {0};
104  pack<Endian, value_bits, word_bits>(value_array, block);
105 
106  // Process the block
107  out = move(c.encrypt(block), out);
108 
109  // Reset seen if we don't need to track the length

111  seen = 0;
112  }
113  }

118  OutputIterator update_n(InputIterator p, size_t n, OutputIterator out) {
119 #ifndef CRYPTO3_BLOCK_NO_OPTIMIZATION
120  for (; n && (seen % block_bits); --n, ++p) {
