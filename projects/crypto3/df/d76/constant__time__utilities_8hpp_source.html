54  /*
55  * T should be an unsigned machine integer type
56  * Expand to a mask used for other operations
57  * @param in an integer
58  * @return If n is zero, returns zero. Otherwise
59  * returns a T with all bits set for use as a mask with
60  * select.
61  */
62  template<typename T>
63  inline T expand_mask(T x) {
64  T r = x;
65  // First fold r down to a single bit
66  for (size_t i = 1; i != sizeof(T) * 8; i *= 2) {
67  r = r | static_cast<T>(r >> i);
68  }
69  r &= 1;
70  r = static_cast<T>(~(r - 1));
71  return r;
72  }
