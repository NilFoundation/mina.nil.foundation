128  inline word_type k(unsigned s, unsigned i) const {
129  word_type x = key_schedule[(s + i) % (key_words + 1)];
130  switch (i) {
131  default:
132  return x;
133  case block_words - 3:
134  return x + tweak_schedule[s % 3];
135  case block_words - 2:
136  return x + tweak_schedule[(s + 1) % 3];
137  case block_words - 1:
138  return x + s;
139  }

142  inline block_type encrypt_block(const block_type &plaintext) const {
143 
144  // Initialize working variables with block
145  block_type v = plaintext;
146 
147 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS
148  for (unsigned t = 0; t < block_words; ++t) {
149  std::printf("v_0,%-2d = %.16lx\n", t, v[t]);
150  }
151 #endif
152 
153  // Encipher block
154  for (unsigned d = 0; d < rounds;) {
155  // Add a subkey (when d%4 == 0)
156  for (unsigned i = 0; i < block_words; ++i) {
157  v[i] += k(d / 4, i);
158 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS
159  std::printf("e_%d,%-2d = %.16lx\n", d, i, v[i]);
160 #endif
161  }
162 
163  // Unrolling by 4 is also useful as the permutations
164  // have a cycle of 2 or 4 (see 8.3)
165  for (unsigned q = 0; q < 4; ++q, ++d) {
166  block_type f = block_type();
167  // MIX into f
168  for (unsigned j = 0; j < block_words / 2; ++j) {
169  word_type x0 = v[2 * j + 0];
170  word_type x1 = v[2 * j + 1];
171  word_type y0 = x0 + x1;
172  std::size_t r = policy_type::rotations[d % 8][j];
173  word_type y1 = policy_type::rotl(x1, r) ^ y0;
174  f[2 * j + 0] = y0;
175  f[2 * j + 1] = y1;
176 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS
177  std::printf("f_%d,%-2d = %.16lx\n", d, 2 * j + 0, f[2 * j + 0]);
178  std::printf("f_%d,%-2d = %.16lx\n", d, 2 * j + 1, f[2 * j + 1]);
179 #endif
180  }
181  // PERMUTE back into v
182  for (unsigned i = 0; i < block_words; ++i) {
183  unsigned pi = policy_type::permutation[i];
184  v[i] = f[pi];
185 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS
186  std::printf("v_%d,%-2d = %.16lx\n", d + 1, i, v[i]);
187 #endif
188  }
189  }
190  }
191 
192  block_type ciphertext = block_type();
193  // Add final subkey
194  for (unsigned i = 0; i < block_words; ++i) {
195  ciphertext[i] = v[i] + k(rounds / 4, i);
196 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS
197  std::printf("c_%-2d = %.16lx\n", i, ciphertext[i]);
198 #endif
199  }
200  return ciphertext;
201  }

203  inline block_type decrypt_block(const block_type &ciphertext) const {
204  for (unsigned i = 0; i < block_words; ++i) {
205 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS
206  std::printf("c_%-2d = %.16lx\n", i, ciphertext[i]);
207 #endif
208  }
209 
210  block_type v = block_type();
211 
212  // Remove final subkey
213  for (unsigned i = 0; i < block_words; ++i) {
214  v[i] = ciphertext[i] - k(rounds / 4, i);
215 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS
216  std::printf("v_%d,%-2d = %.16lx\n", rounds, i, v[i]);
217 #endif
218  }
219 
220  // Decipher block
221  for (unsigned d = rounds; d;) {
222  // Unrolling by 4 is also useful as the permutations
223  // have a cycle of 2 or 4 (see 8.3)
224  for (unsigned q = 4; q--;) {
225  --d;
226 
227  block_type f = block_type();
228  // PERMUTE back into f
229  for (unsigned i = 0; i < block_words; ++i) {
230  unsigned pi = policy_type::permutation[i];
231  f[pi] = v[i];
232 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS
233  std::printf("f_%d,%-2d = %.16lx\n", d, pi, f[pi]);
234 #endif
235  }
236 
237  // UNMIX into v
238  for (unsigned j = 0; j < block_words / 2; ++j) {
239  word_type y0 = f[2 * j + 0];
240  word_type y1 = f[2 * j + 1];
241 
242  std::size_t r = policy_type::rotations[d % 8][j];
243  word_type x1 = policy_type::rotr(y0 ^ y1, r);
244  word_type x0 = y0 - x1;
245 
246  v[2 * j + 0] = x0;
247  v[2 * j + 1] = x1;
248 
249 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS
250  std::printf("e_%d,%-2d = %.16lx\n", d, 2 * j + 0, v[2 * j + 0]);
251  std::printf("e_%d,%-2d = %.16lx\n", d, 2 * j + 1, v[2 * j + 1]);
252 #endif
253  }
254  }
255 
256  // Remove a subkey (when d%4 == 0)
257  for (unsigned i = 0; i < block_words; ++i) {
258  v[i] -= k(d / 4, i);
259 #ifdef CRYPTO3_BLOCK_SHOW_PROGRESS
260  std::printf("d_%d,%-2d = %.16lx\n", d, i, v[i]);
261 #endif
262  }
263  }
264

266  }
