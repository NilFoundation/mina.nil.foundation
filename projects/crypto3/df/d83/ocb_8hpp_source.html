61  const uint8_t *compute_offsets(size_t block_index, size_t blocks) {
62  BOOST_ASSERT_MSG(blocks <= m_max_blocks, "OCB offsets");
63 
64  uint8_t *offsets = m_offset_buf.data();
65 
66  if (block_index % 4 == 0) {
67  const secure_vector<uint8_t> &L0 = get(0);
68  const secure_vector<uint8_t> &L1 = get(1);
69 
70  while (blocks >= 4) {
71  // ntz(4*i+1) == 0
72  // ntz(4*i+2) == 1
73  // ntz(4*i+3) == 0
74  block_index += 4;
75  const size_t ntz4 = ctz<uint32_t>(block_index);
76 
77  xor_buf(offsets, m_offset.data(), L0.data(), m_BS);
78  offsets += m_BS;
79 
80  xor_buf(offsets, offsets - m_BS, L1.data(), m_BS);
81  offsets += m_BS;
82

84  copy_mem(offsets, m_offset.data(), m_BS);
85  offsets += m_BS;
86 
87  xor_buf(m_offset.data(), get(ntz4).data(), m_BS);
88  copy_mem(offsets, m_offset.data(), m_BS);
89  offsets += m_BS;
90 
91  blocks -= 4;
92  }
93  }
94 
95  for (size_t i = 0; i != blocks; ++i) { // could be done in parallel
96  const size_t ntz = ctz<uint32_t>(block_index + i + 1);
97  xor_buf(m_offset.data(), get(ntz).data(), m_BS);
98  copy_mem(offsets, m_offset.data(), m_BS);
99  offsets += m_BS;
100  }
101

103  }

158  void update_nonce(const uint8_t nonce[], size_t nonce_len) {
159  const size_t BS = block_size();
160 
161  const size_t MASKLEN = (BS == 16 ? 6 : ((BS == 24) ? 7 : 8));
162 
163  const uint8_t BOTTOM_MASK = static_cast<uint8_t>((static_cast<uint16_t>(1) << MASKLEN) - 1);
164 
165  secure_vector<uint8_t> nonce_buf(BS);
166 
167  copy_mem(&nonce_buf[BS - nonce_len], nonce, nonce_len);
168  nonce_buf[0] = static_cast<uint8_t>(((tag_size() * 8) % (BS * 8)) << (BS <= 16 ? 1 : 0));
169 
170  nonce_buf[BS - nonce_len - 1] ^= 1;
171 
172  const uint8_t bottom = nonce_buf[BS - 1] & BOTTOM_MASK;
173  nonce_buf[BS - 1] &= ~BOTTOM_MASK;
174 
175  const bool need_new_stretch = (m_last_nonce != nonce_buf);
176 
177  if (need_new_stretch) {
178  m_last_nonce = nonce_buf;
179 
180  m_cipher->encrypt(nonce_buf);
181 
182  /*
183  The loop bounds (BS vs BS/2) are derived from the relation
184  between the block size and the MASKLEN. Using the terminology
185  of draft-krovetz-ocb-wide, we have to derive enough bits in
186  ShiftedKtop to read up to BLOCKLEN+bottom bits from Stretch.
187 
188  +----------+---------+-------+---------+
189  | BLOCKLEN | RESIDUE | SHIFT | MASKLEN |
190  +----------+---------+-------+---------+
191  | 32 | 141 | 17 | 4 |
192  | 64 | 27 | 25 | 5 |
193  | 96 | 1601 | 33 | 6 |
194  | 128 | 135 | 8 | 6 |
195  | 192 | 135 | 40 | 7 |
196  | 256 | 1061 | 1 | 8 |
197  | 384 | 4109 | 80 | 8 |
198  | 512 | 293 | 176 | 8 |
199  | 1024 | 524355 | 352 | 9 |
200  +----------+---------+-------+---------+
201  */
202  if (BS == 16) {
203  for (size_t i = 0; i != BS / 2; ++i) {
204  nonce_buf.push_back(nonce_buf[i] ^ nonce_buf[i + 1]);
205  }
206  } else if (BS == 24) {
207  for (size_t i = 0; i != 16; ++i) {
208  nonce_buf.push_back(nonce_buf[i] ^ nonce_buf[i + 5]);
209  }
210  } else if (BS == 32) {
211  for (size_t i = 0; i != BS; ++i) {
212  nonce_buf.push_back(nonce_buf[i] ^ (nonce_buf[i] << 1) ^
213  (nonce_buf[i + 1] >> 7));
214  }
215  } else if (BS == 64) {
216  for (size_t i = 0; i != BS / 2; ++i) {
217  nonce_buf.push_back(nonce_buf[i] ^ nonce_buf[i + 22]);
218  }
219  }
220 
221  m_stretch = nonce_buf;
222  }
223 
224  // now set the offset from stretch and bottom
225  const size_t shift_bytes = bottom / 8;
226  const size_t shift_bits = bottom % 8;
227 
228  BOOST_ASSERT_MSG(m_stretch.size() >= BS + shift_bytes + 1, "KeyBits ok");
229 
230  secure_vector<uint8_t> offset(BS);
231  for (size_t i = 0; i != BS; ++i) {
232  offset[i] = (m_stretch[i + shift_bytes] << shift_bits);
233  offset[i] |= (m_stretch[i + shift_bytes + 1] >> (8 - shift_bits));
234  }
235 
236  return offset;
237  }

239  static secure_vector<uint8_t> hash(const L_computer<cipher_type> &L,
240  const cipher_type &cipher,
241  const associated_data_type &ad) {
242  const size_t BS = cipher.block_size();
243  secure_vector<uint8_t> sum(BS);
244  secure_vector<uint8_t> offset(BS);
245 
246  secure_vector<uint8_t> buf(BS);
247 
248  const size_t ad_blocks = (ad.size() / BS);
249  const size_t ad_remainder = (ad.size() % BS);
250 
251  for (size_t i = 0; i != ad_blocks; ++i) {
252  // this loop could run in parallel
253  offset ^= L.get(ctz<uint32_t>(i + 1));
254  buf = offset;
255  xor_buf(buf.data(), &ad[BS * i], BS);
256  cipher.encrypt(buf);
257  sum ^= buf;
258  }
259 
260  if (ad_remainder) {
261  offset ^= L.star();
262  buf = offset;
263  xor_buf(buf.data(), &ad[BS * ad_blocks], ad_remainder);
264  buf[ad_remainder] ^= 0x80;
265  cipher.encrypt(buf);
266  sum ^= buf;
267  }
268 
269  return sum;
270  }

308  inline static block_type end_message(const cipher_type &cipher, const block_type &plaintext) {
309  block_type result = {0};
310 
311  const size_t BS = block_size();
312 
313  BOOST_ASSERT_MSG(buffer.size() >= offset, "Offset is sane");
314  const size_t sz = buffer.size() - offset;
315  uint8_t *buf = buffer.data() + offset;
316 
317  secure_vector<uint8_t> mac(BS);
318 
319  if (sz) {
320  const size_t final_full_blocks = sz / BS;
321  const size_t remainder_bytes = sz - (final_full_blocks * BS);
322 
323  encrypt(buf, final_full_blocks);
324  mac = m_L->offset();
325 
326  if (remainder_bytes) {
327  BOOST_ASSERT_MSG(remainder_bytes < BS, "Only a partial block left");
328  uint8_t *remainder = &buf[sz - remainder_bytes];
329 
330  xor_buf(m_checksum.data(), remainder, remainder_bytes);
331  m_checksum[remainder_bytes] ^= 0x80;
332 
333  // Offset_*
334  mac ^= m_L->star();
335 
336  secure_vector<uint8_t> pad(BS);
337  m_cipher->encrypt(mac, pad);
338  xor_buf(remainder, pad.data(), remainder_bytes);
339  }
340  } else {
341  mac = m_L->offset();
342  }
343 
344  // now compute the tag
345 
346  // fold checksum
347  for (size_t i = 0; i != m_checksum.size(); i += BS) {
348  xor_buf(mac.data(), m_checksum.data() + i, BS);
349  }
350 
351  xor_buf(mac.data(), m_L->dollar().data(), BS);
352  m_cipher->encrypt(mac);
353  xor_buf(mac.data(), m_ad_hash.data(), BS);
354 
355  buffer += std::make_pair(mac.data(), tag_size());
356 
357  zeroise(m_checksum);
358  m_block_index = 0;
359 
360  return result;
361  }

419  inline static block_type end_message(const cipher_type &cipher, const block_type &plaintext) {
420  block_type result = {0};
421 
422  const size_t BS = block_size();
423 
424  BOOST_ASSERT_MSG(buffer.size() >= offset, "Offset is sane");
425  const size_t sz = buffer.size() - offset;
426  uint8_t *buf = buffer.data() + offset;
427 
428  BOOST_ASSERT_MSG(sz >= tag_size(), "We have the tag");
429 
430  const size_t remaining = sz - tag_size();
431 
432  secure_vector<uint8_t> mac(BS);
433 
434  if (remaining) {
435  const size_t final_full_blocks = remaining / BS;
436  const size_t final_bytes = remaining - (final_full_blocks * BS);
437 
438  decrypt(buf, final_full_blocks);
439  mac ^= m_L->offset();
440 
441  if (final_bytes) {
442  BOOST_ASSERT_MSG(final_bytes < BS, "Only a partial block left");
443 
444  uint8_t *remainder = &buf[remaining - final_bytes];
445 
446  mac ^= m_L->star();
447  secure_vector<uint8_t> pad(BS);
448  m_cipher->encrypt(mac, pad); // P_*
449  xor_buf(remainder, pad.data(), final_bytes);
450 
451  xor_buf(m_checksum.data(), remainder, final_bytes);
452  m_checksum[final_bytes] ^= 0x80;
453  }
454  } else {
455  mac = m_L->offset();
456  }
457 
458  // compute the mac
459 
460  // fold checksum
461  for (size_t i = 0; i != m_checksum.size(); i += BS) {
462  xor_buf(mac.data(), m_checksum.data() + i, BS);
463  }
464 
465  mac ^= m_L->dollar();
466  m_cipher->encrypt(mac);
467  mac ^= m_ad_hash;
468 
469  // reset state
470  zeroise(m_checksum);
471  m_block_index = 0;

475

478  }
479
