197  char c = static_cast<char>(val);
198  uint8_t retval = 0;
199  if (c >= '0' && c <= '9') {
200  retval = static_cast<uint8_t>(c - '0');
201  } else if (c >= 'A' && c <= 'F') {
202  retval = static_cast<uint8_t>(c - 'A' + 10);
203  } else if (c >= 'a' && c <= 'f') {
