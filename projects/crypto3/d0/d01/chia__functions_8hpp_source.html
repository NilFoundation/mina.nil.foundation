227  inline static uint64_t signed_shift(uint64_t op, int shift) {
228  if (shift > 0) {
229  return op << shift;
230  }
231  if (shift <= -64) {
232  return 0;
233  }
234  return op >> (-shift);
235  }

237  // Return an approximation x of the large mpz_t op by an int64_t and the exponent e adjustment.
238  // We must have (x * 2^e) / op = constant approximately.
239  inline static int64_t mpz_get_si_2exp(signed long int *exp, const mpz_t op) {
240  uint64_t size = mpz_size(op);
241  uint64_t last = mpz_getlimbn(op, size - 1);
242  uint64_t ret;
243  int lg2 = LOG2(last) + 1;
244  *exp = lg2;
245  ret = signed_shift(last, 63 - *exp);
246  if (size > 1) {
247  *exp += (size - 1) * 64;
248  uint64_t prev = mpz_getlimbn(op, size - 2);
249  ret += signed_shift(prev, -1 - lg2);
250  }
251  if (mpz_sgn(op) < 0) {
252  return -((int64_t)ret);
253  }
254  return ret;
255  }
