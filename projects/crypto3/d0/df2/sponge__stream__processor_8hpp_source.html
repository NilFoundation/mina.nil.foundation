169  digest_type end_message() {
170  using namespace nil::crypto3::detail;
171 
172  length_type length = seen;
173 
174  // Add a 1 bit
175 #ifdef CRYPTO3_HASH_NO_OPTIMIZATION
176  std::array<bool, ValueBits> padding_bits = {{1}};
177  std::array<value_type, 1> padding_values;
178  pack<endian, 1, ValueBits>(padding_bits, padding_values);
179  update_one(padding_values[0]);
180 #else
181  value_type pad = 0;
182  imploder_step<endian_type, 1, value_bits, 0>::step(1, pad);
183  update_one(pad);
184 #endif
185 
186  // Pad with 0 bits
187  while ((seen + length_bits) % block_bits != 0) {
188  update_one(value_type());
189  }
190 
191  // Append length
192  append_length<int>(length);
193 
194  // Reset for next message
195  seen = 0;
196 
197  // Calculate static_digest and reset block_hash
198  return block_hash.end_message();
199  }
