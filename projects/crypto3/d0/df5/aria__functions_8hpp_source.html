99  // n-bit right shift of Y XORed to X
100  template<unsigned int N>
101  inline static void rol128(const word_type X[4], const word_type Y[4], word_type KS[4]) {
102  // MSVC is not generating a "rotate immediate". Constify to help it along.
103  static const unsigned int Q = 4 - (N / 32);
104  static const unsigned int R = N % 32;
105  KS[0] = (X[0]) ^ ((Y[(Q) % 4]) >> R) ^ ((Y[(Q + 3) % 4]) << (32 - R));
106  KS[1] = (X[1]) ^ ((Y[(Q + 1) % 4]) >> R) ^ ((Y[(Q) % 4]) << (32 - R));
107  KS[2] = (X[2]) ^ ((Y[(Q + 2) % 4]) >> R) ^ ((Y[(Q + 1) % 4]) << (32 - R));
108  KS[3] = (X[3]) ^ ((Y[(Q + 3) % 4]) >> R) ^ ((Y[(Q + 2) % 4]) << (32 - R));
109  }
