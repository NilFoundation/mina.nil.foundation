423 
424  // CubeHash is only defined for r in {1, 2, 3, ..., 128}
425  BOOST_STATIC_ASSERT(r != 0);

427 
428  // CubeHash is only defined for b in {1, 2, 3, ..., 128}
429  BOOST_STATIC_ASSERT(b != 0);

431 
432  // This implementation of CubeHash only handles b a multiple of 4,

435 
436  // CubeHash is only defined for h in {8, 16, 24, ..., 512}

438  BOOST_STATIC_ASSERT(h <= 512);
439  BOOST_STATIC_ASSERT(h % 8 == 0);
